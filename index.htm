            useEffect(() => {

                if (window.lucide) {

                    window.lucide.createIcons();

                }

            }, [name]);

            return <i data-lucide={name} className={className} style={{ width: size, height: size }}></i>;

        };

            const [selectedLang, setSelectedLang] = useState('English');

            const [searchTerm, setSearchTerm] = useState('');

                    { name: "Aptis ESOL B1", score: "Đạt yêu cầu", provider: "British Council (BC)" },

                    { name: "PEIC Level 2", score: "Pearson English International Certificate", provider: "Pearson" },

                                    {languages.map((lang) => (

                                        <button

                                            key={lang.id}

                                            onClick={() => { setSelectedLang(lang.id); setSearchTerm(''); }}

                                            className={`w-full flex items-center justify-between p-4 rounded-xl transition-all duration-200 group ${

                                                selectedLang === lang.id

                                                ? 'bg-blue-600 text-white shadow-md scale-[1.02]'

                                                : 'bg-white hover:bg-slate-50 text-slate-700 border border-transparent hover:border-slate-200'

                                            }`}

                                        >

                                            <div className="flex items-center gap-4">

                                                <span className="text-2xl">{lang.icon}</span>

                                                <span className="font-bold">{lang.name}</span>

                                            </div>

                                            <Icon name="chevron-right" size={16} className={selectedLang === lang.id ? 'opacity-100' : 'opacity-0 group-hover:opacity-100'} />

                                        </button>

                                    ))}

                                </div>

                            </div>

                           

                            <div className="bg-gradient-to-br from-blue-600 to-indigo-700 rounded-2xl p-6 text-white shadow-lg shadow-blue-200/50">

                                <h3 className="font-bold flex items-center gap-2 mb-2">

                                    <Icon name="award" size={20} /> Điểm 10 Tuyệt Đối

                                </h3>

                                    Tất cả 22 loại chứng chỉ trong danh sách này đều giúp thí sinh đạt điểm tối đa môn Ngoại ngữ trong xét công nhận tốt nghiệp.

                                    value={searchTerm}

                                    onChange={(e) => setSearchTerm(e.target.value)}

                                />

                            </div>

 

                            <div className="grid grid-cols-1 gap-4">

                                {filteredCerts.length > 0 ? (

                                    filteredCerts.map((cert, index) => (

                                        <div key={index} className="bg-white rounded-2xl border border-slate-200 p-6 hover:shadow-lg hover:border-blue-200 transition-all relative overflow-hidden group">

                                                <Icon name="check-circle-2" size={18} className="text-green-500 shrink-0" />

                                                <h3 className="text-xl font-bold text-slate-900 group-hover:text-blue-600 transition-colors">{cert.name}</h3>

                                            </div>

                                            <div className="bg-slate-100 px-3 py-1 rounded inline-block text-slate-600 border border-slate-200 mb-4 text-sm font-medium">

                                                Yêu cầu: <span className="text-blue-600">{cert.score}</span>

                                            </div>

                                            <div className="flex items-start gap-2 text-sm text-slate-500 italic">

                                                <Icon name="file-text" size={16} className="mt-0.5 shrink-0" />

                                                <span>Đơn vị cấp: {cert.provider}</span>

                                            </div>

                                        </div>

                                    ))

                                ) : (

                                    <div className="text-center py-20 bg-white rounded-3xl border-2 border-dashed border-slate-200">

                                        <h3 className="text-slate-900 font-bold text-lg">Không tìm thấy chứng chỉ</h3>

                                        <p className="text-slate-500">Hãy thử từ khóa khác.</p>

                                    </div>

                                )}

                            </div>

                        </div>

                    </div>

                   

                    <footer className="mt-12 text-center text-slate-400 text-sm">

                        <p>© 2026 Bản quyền dữ liệu thuộc Bộ Giáo dục và Đào tạo Việt Nam</p>

                    </footer>

                </div>

            );

        };

 

        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>
